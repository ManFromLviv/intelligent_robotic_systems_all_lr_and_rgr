#include <Servo.h>  // Підключаємо бібліотеку для роботи з сервомоторами

Servo servo1;  // Створюємо об'єкт сервомотора з ім'ям servo1

int PIN = 6;  // Змінна для зберігання номера цифрового піну, до якого підключено сервомотор
int MIN = 500;  // Мінімальна ширина імпульсу (в мікросекундах), яка відповідає мінімальному куту
int MAX = 2500;  // Максимальна ширина імпульсу (в мікросекундах), яка відповідає максимальному куту

int potentiometer_value;  // Змінна для збереження значення з потенціометра
int angle;  // Змінна для збереження обчисленого кута для сервомотора
int potentiometer_pin = 0;  // Аналоговий пін, до якого підключено потенціометр (A0)

void setup() {
  servo1.attach(PIN, MIN, MAX);  // Прив'язуємо сервомотор до піну та задаємо діапазон сигналу (мікросекунди)
}

void loop() {
  potentiometer_value = analogRead(potentiometer_pin);  // Зчитуємо аналогове значення з потенціометра (0–1023)
  angle = map(potentiometer_value, 0, 1023, 0, 180);  // Перетворюємо значення потенціометра на кут (0–180 градусів)
  servo1.write(angle);  // Встановлюємо відповідний кут сервомотору
  delay(20);  // Коротка затримка для стабільної роботи сервомотора
}
